---
title: "Build to Stock"
api: "POST https://api.stateset.com/api/billofmaterials/:id/buildtostock"
description: "This endpoint updates the status of a bill of materials to build to stock."
---

### Body

<ParamField body="bill_of_materials_id" type="string">
  The ID provided in the data tab may be used to identify the bill_of_materials
</ParamField>

### Response

<ResponseField name="id" type="string">
  The ID provided in the data tab may be used to identify the return
</ResponseField>

<ResponseField name="object" type="string">
  The object type
</ResponseField>

<ResponseField name="success" type="number">
  Indicates whether the call was successful. true if successful, false if not.
</ResponseField>

    "id": "e0901f083-aa1c-43c5-af5c-0a9d2fc64e30"

const buildToStock = await stateset.billofmaterials.buildToStock({
